Services Example

myModule.service('Developer', function () {
  this.name = 'Foo';
  this.motherLanguage = 'JavaScript';
  this.code = function () {};
  this.live = function () {
    while (true) {
      this.code();
    }
  };
});
function MyCtrl(Developer) {
  var developer = new Developer();
  developer.live();
}
